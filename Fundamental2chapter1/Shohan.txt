conn sys as sysdba;
system

#For create user managers:

create user managers identified by managers;
grant dba to managers;
create table coderbd(id number(10),name varchar2(50));


#For create user operators: 

create user operators identified by operators;
grant create session to operators;
grant select,insert on coderbd to operators with grant option;


#For create user guests:

create user guests identified by guests; 
grant create session to guests; 
grant select on coderbd to guest with grant option;


#Connect operators (for select ,insert):
conn operators/operators;
insert in to managers.coderbd(id,name) values(100,'shamim');

#Connect guests(for select):
conn guests/guests;
select * from managers.coderbd;
